Core FastAPI Handler (Python)
# FastAPI + YOLOv8 + Twilio — WildGuard Cloud Backend
from fastapi import FastAPI, UploadFile
from ultralytics import YOLO
from twilio.rest import Client
import cv2, numpy as np

app = FastAPI()
model = YOLO("wildlife_yolov8n.pt")  # fine-tuned weights
twilio = Client(ACCOUNT_SID, AUTH_TOKEN)

THREAT_MATRIX = {
    "elephant": "HIGH", "gaur": "HIGH", "leopard": "HIGH",
    "wild_boar": "MEDIUM", "deer": "LOW", "monkey": "LOW",
}

@app.post("/api/detect")
async def detect(image: UploadFile):
    img_bytes = await image.read()
    arr = np.frombuffer(img_bytes, np.uint8)
    frame = cv2.imdecode(arr, cv2.IMREAD_COLOR)

    results = model(frame)[0]
    animal = results.names[int(results.boxes.cls[0])]
    confidence = float(results.boxes.conf[0])
    threat = THREAT_MATRIX.get(animal, "LOW")

    gpio = {"HIGH": {"buzzer":1,"strobe":1,"sms":1},
            "MEDIUM": {"buzzer":1,"strobe":0,"sms":1},
            "LOW": {"buzzer":0,"strobe":0,"sms":0}}[threat]

    if gpio["sms"]:
        twilio.messages.create(
            body=f"⚠️ {animal.upper()} detected — {threat} THREAT!",
            from_=TWILIO_NUMBER, to=FARMER_NUMBER)

    return {"animal": animal, "threat_level": threat,
            "confidence": round(confidence*100),
            "gpio_commands": gpio}
